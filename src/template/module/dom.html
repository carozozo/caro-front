dom
簡介
DOM 選取器, 支援一些方便的程式
參數
$('xxx').dom([cb]): 賦予本身 dom 屬性
$('xxx').dom(selector, [cb]): 相當於.find(selector), 並賦予 dom 屬性
selector: 目標選擇器
cb: 呼叫之後觸發的 fn($self), $self 就是 DOM 本身
程式範例
<div id='#xx'> <div id='#yy'> </div> </div>
var $xx = $('#xx').dom();
var id = $xx.id(); // 'xx'
var $yy = $xx.dom('#yy');
var id = $yy.id(); // 'yy'
函式
id([id]): 設置/取得 id
name([name]): 設置/取得 name
aClass([class]): 設置/取得 class
title([title]): 設置/取得 title
src([src]): 設置/取得 src
href([href, target]): 設置/取得 href
ifVisible(): 是否為可見
ifHidden(): 是否為隱藏
enable(): 設為 enable
disable(): 設為 disable
sethecked([checked = true]): 設為 checked / unchecked
ifChecked(): 是否為 checked
getCssDistance(cssType): 取得 css 設置的 top/left/right/bottom...等距離程式範例
假設 $('#xx') 的 css top: 10px
$('#xx').dom().getCssDistance('top'); // 10
$('#xx').css('top'); // '10px'
參數
cssType: 'top'/'left'/'right'/'bottom'/'margin-left'....
getCssWidth(): 取得 css 的 width string程式範例
假設 $('#xx') 的 css width: 100%
$('#xx').dom().getCssWidth(); // '100%'
$('#xx').css('width'); // '100px'
getCssHeight(): 取得 css 的 height string程式範例
假設 $('#xx') 的 css height: 100%
$('#xx').dom().getCssHeight(); // '100%'
$('#xx').css('height'); // '100px'
setCursor([cursor = 'pointer']): 設置滑鼠指標
blurUpperCase([nameSpace]): 觸發 blur 後, 將裡面的值 String.trim() 之後轉大寫
action(eventName, fn): 同等 $('xxx').off('event').on('event',function(){}), 避免重複綁定
onClick(fn, [nameSpace]): $('xxx').off('click').on('click') 方便使用版, 滑鼠移到上面會自動轉成手指
onPressEnter(fn, [nameSpace]): 按下 Enter 鍵後觸發的 fn, 綁定 keyup
eachDom(cb):.each() 進階版, 直接賦予 dom 屬性程式範例
$('.xx').eachDom(function($dom, i){
// $dom 相當於 $(this), 並且可以調用 dom 的屬性
$dom.onClick(...);
});
參數
cb: 呼叫之後觸發的 fn($dom, index)
mapDom(cb):$.map() 進階版, 直接賦予 dom 屬性, 並回傳 dom array程式範例
// $domList 相當於 $('.xx') 取得的每個 DOM 組成的 array
var $domList = $('.xx').mapDom(function($dom, i){
// $dom 相當於 $(this), 並且可以調用 dom 的屬性
$dom.onClick(...);
});
參數
cb: 呼叫之後觸發的 fn($dom, index)
ifEmpty(): 判斷本身是否沒任何 html 內容程式範例
<div id='#xx'>caro-front</div>
$('.xx').dom().ifEmpty(); // false
<div id='#xx'></div>
$('.xx').dom().ifEmpty(); // true
getHtml(): 取得包含本身的 html code程式範例
<div id='#xx'>caro-front</div>
$('.xx').dom().getHtml(); // <div id='#xx'>caro-front</div>
$('.xx').html(); // caro-front

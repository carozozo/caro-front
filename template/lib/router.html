簡介
換頁函式庫, 使用 $.ajax 切換分頁, 並利用 url hash 記錄
設定檔
templateDir: 放置分頁的資料夾路徑, 未設置代表根目錄
templateExtname: 放置的分頁路徑副檔名, 例如 html, php, jsp
container: 分頁容器的 DOM id, 未設置時預設容器為 $('body')
屬性
$container: 當前分頁的容器 DOM 
$page: 當前分頁 DOM
_befPage: 儲存每次執行 cf.goPage 前要執行的 fns
_prePage: 紀錄載入分頁前會執行的 fns 
_page: 當載入分頁後要執行的指定 fn  
_aftPage: 紀錄載入分頁後會執行的 fn s
_transitionFn: 紀錄當前換頁效果 fn 
pageName: 當前分頁名稱 
templateDir: 記錄要放置分頁的資料夾路徑 
templateExtname: 記錄要載入的分頁副檔名
函式
getPageByHash([hash]): 依照 url 的 hash 判斷目前的分頁路徑名稱 範例
假設現在的 url 是 https://carozozo.github.io/caro-front/#a/b?search=2
cf.getPageByHash();  // 'a/b'
cf.getPageByHash('#router?search=1');  // 'router'
getSearchByHash([hash]): 依照 url 的 hash 判斷目前的 search 範例
假設現在的 url 是 https://carozozo.github.io/caro-front/#index?name=caro&age=20
cf.getSearchByHash();  // 'name=caro&age=20'
cf.getSearchByHash('#router?name=huang');  // 'name=huang'
parseUrlSearch(): 依照 url 的 hash 裡面的 search 轉為 obj 範例
假設現在的 url 是 https://carozozo.github.io/caro-front/#index?name=caro&age=20
cf.getSearchByHash();  // {name: 'caro', age: '20'}
cf.getSearchByHash('#router?name=huang');  // {name: 'huang'}
regBefPage(fn, [執行順序=50]): 註冊每次執行 cf.goPage 前要執行的 fn 範例
// 每次執行 cf.goPage 前, 會先 alert 'b', 然後 alert 'a'
cf.regBefPage(function(cf){
  alert('a');
});
cf.regBefPage(function(cf){
  alert('b');
}, 1);
regPrePage(fn, [執行順序=50]): 註冊每次載入分頁前要執行的 fn 範例
// 每次載入分頁前, 會先 alert 'b', 然後 alert 'a'
cf.regPrePage(function(cf, $page){   
// $page 就是分頁的 jQuery DOM 物件
  alert('a');
});
cf.regPrePage(function(cf, $page){   
// $page 就是分頁的 jQuery DOM 物件
  alert('b');
}, 1);
regAftPage(fn, [執行順序=50]): 註冊每次載入分頁後要執行的 fn 範例
// 每次載入分頁後, 會先 alert 'b', 然後 alert 'a'
cf.regAftPage(function(cf, $page){   
// $page 就是分頁的 jQuery DOM 物件
  alert('a');
});
cf.regAftPage(function(cf, $page){   
// $page 就是分頁的 jQuery DOM 物件
  alert('b');
}, 1);
regPage(註冊分頁, fn): 註冊載入指定分頁後要執行的 fn 範例
// 執行 cf.goPage('page1') 時, 會 alert 'a'
cf.regPage('page1', function(cf, $page){   
// $page 就是分頁的 jQuery DOM 物件
  alert('b');
});
// 執行 cf.goPage('page2') 時, 會 alert 'b'
cf.regPage('page2', function(cf, $page){   
// $page 就是分頁的 jQuery DOM 物件
  alert('b');
});
goPage([分頁路徑], [opt={}]): 換頁, 沒參數時會依據 url 的 hash 自動判斷 範例
opt 參數
fileType: 要載入的分頁副檔名
設定檔 templateDir = 'temp', templateExtname = 'jsp'
// 讀取 /temp/page1.jsp 裡的內容並寫入容器
cf.goPage('page1');
// 讀取 /temp/subPage/page1.jsp 裡的內容並寫入容器
cf.goPage('subPage/page1');
// 讀取 /temp/subPage/page2.html 裡的內容並寫入容器
cf.goPage('subPage/page2', {fileType: 'html'});
設定檔 templateDir = '', templateExtname = ''
// 讀取 /subPage/page2 裡的內容並寫入容器
cf.goPage('subPage/page2');
blockGoPage(): 呼叫後, 執行 router.goPage 不會換頁 範例
cf.blockGoPage();
// 不會換頁
cf.goPage('index');
approveGoPage(): 呼叫後, 執行 router.goPage 可以換頁   範例
cf.blockGoPage();
// 不會換頁
cf.goPage('index');

cf.approveGoPage();
// 會換頁
cf.goPage('index');
goPage 執行流程
執行儲存在 _befPage 裡的 fns
取得要載入的分頁名稱
透過 ajax 從 template 取得對應的分頁內容
執行儲存在 _prePage 裡的 fns
移除原本的分頁內容, 並置入目標分頁內容
執行對應的 _page fn
執行儲存在 _aftPage 裡的 fns

html
<div id='target'>a: {{a}}, b: {{b}}<div/>

js
var obj = {a: 1, b: 2}
$('#setBtn1').on('click', function() {
$('#target').cfSyncObjVal(obj);
});
$('#setBtn2').on('click', function() {
obj.a = 'Caro';
});
